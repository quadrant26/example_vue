 elementUI 安装
	
	1. 安装

		npm i element-ui -D
			
			i 	-> 	install

			D 	-> 	--save-dev

			S 	-> 	--save
	
	2. 使用

		在入口文件引入

			import ElementUI from 'element-ui'
			import 'element-ui/lib/theme-default/index.css'
			
			Vue.use(ElementUI)
